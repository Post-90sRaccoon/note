function forof(generator, action) {
  var generated = generator.next()
  while (!generated.done) {
    if(action(generated.value) === false){
      generator.return() //把生成器关掉
      break
    }
    generated = generator.next()
  }
}

forof(f(), (x) => {
  console.log(x)
  return false
})

```javascript
Number.prototype.digits = function* () {
  var n = this.valueOf()
  if (n == 0) {
    yield 0
    return
  }
  while (n > 0) {
    var digit = n % 10
    yield digit
    n = (n - digit) / 10
  }
}

for (var x of 10086..digits()) {
  console.log(x)
}



Number.prototype.digits = function* () {
  var n = this.valueOf()

  var digit = n % 10
  var rest = (n - digit) / 10
  if (rest > 0) {
    yield* rest.digits()
  }
  yield digit
}

for (var x of 10086..digits()) {
  console.log(x)
}
```
